create table Passenger(
Id integer not null constraint PK_Passenger primary key,
FirstName varchar(20) not null,
LastName varchar(20) not null,
DateOfBirth date not null
);

create table Ticket(
Id integer not null constraint PK_Ticket primary key,
Price float not null,
TicketDate timestamp not null,
PassengerId integer not null constraint FK_Passenger_PassengerId references Passenger(Id),
RouteId integer not null constraint FK_Route_RouteId references Route(Id),
VehicleId integer not null constraint FK_Vehicle_VehicleId references Vehicle(Id)
);

create table Route(
Id integer not null constraint PK_Route primary key,
RouteName varchar(20) not null,
NumberOfStops integer not null
);

create table Vehicle(
Id integer not null constraint PK_Vehicle primary key,
VehicleModel varchar(20) not null,
Capacity integer not null
);

create table RouteVehicle(
RouteId integer not null constraint FK_Route_RouteId references Route(Id),
VehicleId integer not null constraint FK_Vehicle_VehicleId references Vehicle(Id),
constraint PK_Vehicle_Route primary key (RouteId, VehicleId)
);

create table VehicleRegistration(
VehicleId integer not null constraint FK_VehicleRegistration_Vehicle_VehicleId references Vehicle(Id), 
StartDate date not null,
EndDate date not null);
